164  template<typename base_type>
165  base_type intpow(const base_type base, const int exponent) {
166  base_type result;
167  if (exponent == 0) {
168  result = 1;
169  return result;
170  }
171  int exponent_abs;
172  if (exponent < 0) {
173  exponent_abs = -exponent;
174  }
175  else {
176  exponent_abs = exponent;
177  }
178  result = base;
179  for (int i = 1; i < exponent_abs; i++) {
180  result *= base;
181  }
182  if (exponent < 0) {
183  result = 1.0 / result;
184  }
185  return result;
186  }

191  template<typename T>
192  void removeDuplicates(vector<T>& vec) {
193  for (auto it1 = vec.begin(); it1 < vec.end() - 1; ++it1) {
194  for (auto it2 = it1 + 1; it2 < vec.end(); ++it2) {
195  if (*it2 == *it1) {
196  *it2 = *(it2 - 1);
197  }
198  }
199  }
200  auto it3 = unique(vec.begin(), vec.end());
201  vec.resize(distance(vec.begin(), it3));
202  }
